--数据库
CREATE SCHEMA `resourcesnetwork` DEFAULT CHARACTER SET utf8 COLLATE utf8_bin ;

-- 管理员表
DROP TABLE IF EXISTS `ADMIN`;
CREATE TABLE `ADMIN` (
  `ID` INT NOT NULL,
  `NAME` VARCHAR(45) NULL,
  `LOGINNAME` VARCHAR(45) NULL,
  `PASSWARD` VARCHAR(45) NULL,
  `STOPED` INT(1) NULL,
  `SALT` INT(5) NULL,
  PRIMARY KEY (`ID`),
  UNIQUE INDEX `ID_UNIQUE` (`ID` ASC)
) ENGINE=InnoDB DEFAULT CHARSET=utf8 COLLATE=utf8_bin;

-- 角色表
DROP TABLE IF EXISTS `role`;
CREATE TABLE `role` (
  `ROLE_ID` int(11) NOT NULL,
  `ROLE_NAME` varchar(45) COLLATE utf8_bin DEFAULT NULL,
  `ROLE_DES` varchar(45) COLLATE utf8_bin DEFAULT NULL,
  `STOPED` int(1) DEFAULT NULL,
  PRIMARY KEY (`ROLE_ID`),
  UNIQUE KEY `ROLE_ID_UNIQUE` (`ROLE_ID`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8 COLLATE=utf8_bin;

-- 权限表
DROP TABLE IF EXISTS `permission`;
CREATE TABLE `permission` (
  `PERMISSION_ID` INT NOT NULL,
  `PERMISSION_NAME` VARCHAR(45) NULL,
  `PERMISSION_DES` VARCHAR(45) NULL,
  `STOPED` INT(1) NULL,
  PRIMARY KEY (`PERMISSION_ID`),
  UNIQUE INDEX `Permiss_id_UNIQUE` (`PERMISSION_ID` ASC)
) ENGINE=InnoDB DEFAULT CHARSET=utf8 COLLATE=utf8_bin;

-- 管理员角色对应表
DROP TABLE IF EXISTS `admin_role`;
CREATE TABLE `admin_role` (
  `ADMIN_ID` INT NOT NULL,
  `ROLE_ID` INT NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8 COLLATE=utf8_bin;

-- 角色权限对应表
DROP TABLE IF EXISTS `role_permission`;
CREATE TABLE `role_permission` (
  `ROLE_ID` INT NOT NULL,
  `PERMISSION_ID` INT NULL
) ENGINE=InnoDB DEFAULT CHARSET=utf8 COLLATE=utf8_bin;
